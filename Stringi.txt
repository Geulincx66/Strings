import java.io.File;
import java.io.FileInputStream;
import java.io.PrintWriter;
import java.io.FileNotFoundException;
import java.util.Scanner;


public class Main 
{

	public static String firstUp(String s) {
		String[] words = s.split(" ");
		String result = "";

		for (String word : words) {
			String first = word.substring(0,1);
			first = first.toUpperCase();  
			result +=  first + word.substring(1)+" ";

		}
		return result;
	}
	
	public static String secondUp(String s)
	{	
		String result = "";
		int index = 0;
		
		char[] tabOfChars = s.toCharArray();
		for (char c : tabOfChars) 
		{
		if(index % 2 == 0)
		{
		result = result + String.valueOf(c).toUpperCase();
		}
		 
		else
		{
		result = result + String.valueOf(c);
		}
	
		index++;
		}
		return result;
	
}
	
	public static String reverse(String s){
	    char[] in = s.toCharArray();
	    int begin=0;
	    int end=in.length-1;
	    char temp;
	    while(end>begin){
	        temp = in[begin];
	        in[begin]=in[end];
	        in[end] = temp;
	        end--;
	        begin++;
	    }
	    return new String(in);
	}
	
	
	public static void main(String[] args) throws FileNotFoundException
	{
		File file = new File("plik.txt");
		Scanner odczyt = new Scanner(file);
		
		String line = odczyt.nextLine();
		
		System.out.println(line);
		//System.out.println(firstUp(line)); //Zamiana kazdej pierwszej litery na wielka.
		System.out.println(line.toUpperCase()); //Zamiana wszystkich liter na wielkie
		//System.out.println(secondUp(line)); //Zamiana co drugiej litery na wielka
		System.out.println(reverse(line)); //Odwracanie stringa.
		
		PrintWriter zapis = new PrintWriter("zapis.txt");
		String line2=reverse(line);
		zapis.println(line2);
		
		odczyt.close();
		zapis.close();
		

	}

}
